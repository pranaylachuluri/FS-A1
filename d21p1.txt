class TreeNode
{
    public int data;
    public TreeNode left, right;
    public TreeNode(int data)
    {
        this.data = data;
        left = null;
        right = null;
    }

import java.util.*;
class Solution
{
    int sum=0;
    public int sumOfLeftLeaves(TreeNode root)
    {
        //Implement Your Code here
        TreeNode curr=root;
        if(curr==null){
            return sum;
        }
        if(curr.left!=null){
            sum+=curr.left.data;
            sumOfLeftLeaves(curr.left);
            sumOfLeftLeaves(curr.right);
        }
        return sum;
    }
}
